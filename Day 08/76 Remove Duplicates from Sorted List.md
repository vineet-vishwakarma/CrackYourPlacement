# Code
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head==NULL)
            return NULL;
        
        ListNode*i=head;
        ListNode*j=head->next;

        while(i && j){
            if(i->val!=j->val){
                i->next=j;
                i=i->next;
            }
            j=j->next;
        }
        i->next=NULL;
        return head;
    }
};
```
